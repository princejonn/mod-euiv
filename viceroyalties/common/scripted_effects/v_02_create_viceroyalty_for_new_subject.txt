create_viceroyalty_for_new_subject = {
	# Cede Provinces
	cede_all_provinces_in_sector_fn = {
		SECTOR = $SECTOR$
		OVERLORD = $OVERLORD$
		SUBJECT = $SUBJECT$
	}
	# Set Government Type
	if = {
		limit = {
			$OVERLORD$ = {
				government = monarchy
			}
		}
		$SUBJECT$ = {
			change_government = monarchy
		}
	}
	else_if = {
		limit = {
			$OVERLORD$ = {
				government = republic
			}
		}
		$SUBJECT$ = {
			change_government = republic
		}
	}
	# Set Government Rank
	if = {
		limit = {
			$OVERLORD$ = {
				government_rank = 3
			}
		}
		$SUBJECT$ = {
			set_government_rank = 2
		}
	}
	else_if = {
		limit = {
			$OVERLORD$ = {
				government_rank = 2
			}
		}
		$SUBJECT$ = {
			set_government_rank = 2
		}
	}
	else = {
		$SUBJECT$ = {
			set_government_rank = 1
		}
	}
	# Set Opinion and Religion
	$SUBJECT$ = {
		add_historical_friend = $OVERLORD$
		change_religion = $OVERLORD$
	}
	# Create Subject
	create_subject = {
		subject = $SUBJECT$
		subject_type = viceroyalty
		keep_liberty_desire = yes
	}
}
