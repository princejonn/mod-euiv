create_viceroyalty_from_existing_subject = {
	# Cede Provinces
	cede_all_provinces_to_subject = {
		OVERLORD = $OVERLORD$
		OWNER = $OWNER$
		SUBJECT = $SUBJECT$
	}
	# Change Government
	change_subject_government_fn = {
		SUBJECT = $SUBJECT$
		OVERLORD = $OVERLORD$
	}
	# Create Subject
	create_subject = {
		subject = $SUBJECT$
		subject_type = viceroyalty
		keep_liberty_desire = yes
	}
	# Power and Prestige
	add_dip_power = -120
	add_prestige = 10
}
