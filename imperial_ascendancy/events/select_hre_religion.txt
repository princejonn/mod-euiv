namespace = select_hre_religion
country_event = {
	id = select_hre_religion.1
	is_triggered_only = yes
	hidden = yes
	immediate = {
		if = {
			limit = {
				emperor = {
					religion = catholic
				}
			}
			country_event = {
				id = select_hre_religion.2
			}
		}
		else_if = {
			limit = {
				emperor = {
					religion = protestant
				}
			}
			country_event = {
				id = select_hre_religion.3
			}
		}
		else = {
			country_event = {
				id = select_hre_religion.4
			}
		}
	}
	option = {
		ai_chance = {
			factor = 10
		}
	}
}

# Catholicism
country_event = {
	id = select_hre_religion.2
	title = select_hre_religion.2.t
	desc = select_hre_religion.2.d
	picture = RELIGIOUS_WARS_eventPicture
	major = yes
	is_triggered_only = yes
	immediate = {
		set_hre_religion_treaty = yes
		set_hre_heretic_religion = protestant
		set_hre_religion = catholic
		set_hre_religion_locked = yes
	}
	option = {
		name = select_hre_religion.2.o
		tooltip = {
			set_hre_religion_treaty = yes
			set_hre_heretic_religion = protestant
			set_hre_religion = catholic
			set_hre_religion_locked = yes
		}
	}
}
