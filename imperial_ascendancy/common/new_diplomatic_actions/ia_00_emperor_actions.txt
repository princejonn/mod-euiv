emperor_force_release_nations = {
	category = emperor
	require_acceptance = no
	is_visible = {
		ai = no
		is_emperor = yes
		is_part_of_hre = yes
		FROM = {
			ai = yes
			is_part_of_hre = yes
		}
	}
	is_allowed = {
		imperial_influence = 25
		is_subject_of = emperor
		OR = {
			hre_reform_passed = emperor_landfriede
			hre_reform_passed = emperor_privilegia_de_non_appelando
		}
		FROM = {
			any_province = {
				is_part_of_hre = yes
				any_core_country = {
					exists = no
				}
			}
		}
	}
	on_accept = {
		add_imperial_influence = -25
		FROM = {
			release_all_possible_countries = yes
			add_opinion = {
				who = ROOT
				modifier = forced_release_nations
			}
		}
	}
	on_decline = {
	}
	ai_will_do = {
		always = no
	}
}
