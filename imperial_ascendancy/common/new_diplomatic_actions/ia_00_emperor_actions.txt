emperor_force_release_nations = {
	category = emperor
	require_acceptance = no
	is_visible = {
		ai = no
		is_emperor = yes
		is_part_of_hre = yes
		FROM = {
			ai = yes
			is_part_of_hre = yes
		}
	}
	is_allowed = {
		imperial_influence = 25
		hre_reform_passed = emperor_reichsstabilitaet
		FROM = {
			has_opinion = {
				who = ROOT
				value = 150
			}
		}
		custom_trigger_tooltip = {
			tooltip = target_has_releasable_nations_tooltip
			FROM = {
				any_province = {
					is_part_of_hre = yes
					any_core_country = {
						exists = no
					}
				}
			}
		}
		custom_trigger_tooltip = {
			tooltip = emperor_has_more_cities_than_target_tooltip
			num_of_cities = FROM
			NOT = {
				FROM = {
					num_of_cities = ROOT
				}
			}
		}
	}
	on_accept = {
		add_imperial_influence = -25
		FROM = {
			release_all_possible_countries = yes
			add_opinion = {
				who = ROOT
				modifier = forced_release_nations
			}
		}
	}
	on_decline = {
	}
	ai_will_do = {
		always = no
	}
}
