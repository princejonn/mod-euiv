is_hre_guarantee_member = {
	is_part_of_hre = yes
	is_subject = no
	NOT = {
		tag = BOH
	}
	NOT = {
		tag = MLO
	}
}

num_emperor_guarantees = {
	calc_true_if = {
		all_known_country = {
			is_part_of_hre = yes
			OR = {
				guaranteed_by = emperor
				emperor = {
					is_supporting_independence_of = PREV
				}
			}
		}
		amount = $amount$
	}
}
