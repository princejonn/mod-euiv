give_income = {
	add_years_of_income = $amount$
}

give_land_share = {
	change_estate_land_share = {
		estate = all
		share = $amount$
	}
}

give_legitimacy = {
	if = {
		limit = {
			government = monarchy
		}
		add_legitimacy = $amount$
	}
	else_if = {
		limit = {
			government = republic
		}
		add_republican_tradition = $amount$
	}
	else_if = {
		limit = {
			government = theocracy
		}
		add_devotion = $amount$
	}
}

give_loyalty = {
	add_estate_loyalty = {
		estate = all
		loyalty = $amount$
	}
}

give_power = {
	add_adm_power = $amount$
	add_dip_power = $amount$
	add_mil_power = $amount$
}

give_tradition = {
	add_army_tradition = $amount$
	add_navy_tradition = $amount$
}

give_yearly_stipend = {
	if = {
		limit = {
			ai = no
			has_country_flag = enabled_yearly_stipend_country_flag
		}
		if = {
			limit = {
				has_country_flag = yearly_stipend_flag_1
			}
			give_yearly_stipend_1 = yes
		}
		else_if = {
			limit = {
				has_country_flag = yearly_stipend_flag_2
			}
