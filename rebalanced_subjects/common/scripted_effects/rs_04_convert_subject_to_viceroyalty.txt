convert_subject_to_viceroyalty_fn = {
	hidden_effect = {
		# Change Tag
		change_subject_tag_fn = {
			SUBJECT = $SUBJECT$
			NEW_TAG = $NEW_TAG$
		}
		# Cede Provinces
		cede_all_owned_provinces_in_sector_fn = {
			SECTOR = $SECTOR$
			OWNER = $OVERLORD$
			RECEIVER = $NEW_TAG$
		}
		cede_all_owned_provinces_outside_sector_fn = {
			SECTOR = $SECTOR$
			OWNER = $NEW_TAG$
			RECEIVER = $OVERLORD$
		}
		# Permanent Claims
		add_permanent_claims_in_sector_fn = {
			SECTOR = $SECTOR$
			OWNER = $NEW_TAG$
		}
		# Change Government
		change_subject_government_fn = {
			SUBJECT = $NEW_TAG$
			OVERLORD = $OVERLORD$
		}
		# Change Name
		change_subject_name_fn = {
			SUBJECT = $NEW_TAG$
			SECTOR = $SECTOR$
		}
	}
	# Create Subject
	create_subject = {
		subject_type = viceroyalty
		subject = $NEW_TAG$
		keep_liberty_desire = yes
	}
	$NEW_TAG$ = {
		capital_scope = {
			add_unrest = 1
		}
		add_liberty_desire = 10
		add_stability = -1
	}
}
