stop_subject_bickering = {
	every_country = {
		limit = {
			ai = no
		}
		save_event_target_as = bickering_overlord_target
		every_subject_country = {
			save_event_target_as = bickering_subject_target
			every_claimed_province = {
				limit = {
					OR = {
						owned_by = event_target:bickering_overlord_target
						owned_by_subject_of = event_target:bickering_overlord_target
					}
				}
				save_event_target_as = bickering_province_target
				event_target:bickering_province_target = {
					remove_claim = event_target:bickering_subject_target
				}
			}
		}
	}
}
