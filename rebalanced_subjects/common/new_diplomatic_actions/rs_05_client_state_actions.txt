form_client_state_from_vassal = {
	category = influence
	require_acceptance = no
	is_visible = {
		FROM = {
			is_subject_of = ROOT
			is_subject_of_type = vassal
		}
	}
	is_allowed = {
		is_at_war = no
		full_idea_group = influence_ideas
		NOT = {
			truce_with = FROM
		}
		FROM = {
			has_opinion = {
				who = ROOT
				value = 190
			}
		}
		FROM = {
			NOT = {
				liberty_desire = 10
			}
		}
		FROM = {
			years_in_vassalage_under = {
				who = ROOT
				years = 10
			}
		}
	}
	on_accept = {
		create_subject = {
			subject_type = client_state
			subject = FROM
			keep_liberty_desire = yes
		}
		FROM = {
			capital_scope = {
				add_unrest = 2
			}
			add_liberty_desire = 10
			add_stability = -1
		}
	}
	on_decline = {
	}
	ai_will_do = {
		always = no
	}
}
