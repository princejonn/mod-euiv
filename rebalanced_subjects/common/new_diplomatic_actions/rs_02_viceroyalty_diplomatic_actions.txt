convert_to_viceroyalty = {
	category = influence
	require_acceptance = no
	is_visible = {
		adm_tech = 15		#Military Administration
		ai = no
		is_subject = no
		technology_group = western
		FROM = {
			OR = {
				is_subject_of_type_with_overlord = {
					who = ROOT
					type = vassal
				}
				is_subject_of_type_with_overlord = {
					who = ROOT
					type = incorporated_vassal
				}
			}
			capital_scope = {
				is_province_in_sector_switch = yes
			}
		}
	}
	is_allowed = {
		adm_power = 75
		dip_power = 75
		is_at_war = no
		NOT = {
			truce_with = FROM
		}
		custom_trigger_tooltip = {
			tooltip = TOOLTIP_CANNOT_ESTABLISH_VICEROYALTY_FROM_VASSAL
			FROM = {
				can_convert_to_viceroyalty_switch = yes
			}
		}
		FROM = {
			is_at_war = no
		}
		FROM = {
			is_march = no
		}
		FROM = {
			NOT = {
				truce_with = ROOT
			}
		}
		FROM = {
			years_in_vassalage_under = {
				who = ROOT
				years = 25
			}
		}
		FROM = {
			has_opinion = {
				who = ROOT
				value = 190
			}
		}
		FROM = {
			NOT = {
				liberty_desire = 50
			}
		}
		OR = {
			NOT = {
				has_country_flag = establish_viceroyalty_cooldown_flag
			}
			had_country_flag = {
				flag = establish_viceroyalty_cooldown_flag
				days = 730
			}
		}
	}
	on_accept = {
		# Convert Subject
		convert_subject_to_viceroyalty_switch = {
			OVERLORD = ROOT
			SUBJECT = FROM
		}
		# Power and Prestige
		add_adm_power = -75
		add_dip_power = -75
		add_prestige = 20
		# Flags
		clr_country_flag = establish_viceroyalty_cooldown_flag
		set_country_flag = establish_viceroyalty_cooldown_flag
	}
	ai_will_do = {
		always = no
	}
}
