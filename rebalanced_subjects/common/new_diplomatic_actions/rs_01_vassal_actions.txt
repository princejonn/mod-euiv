incorporate_vassal = {
	category = influence
	require_acceptance = no
	is_visible = {
		OR = {
			can_incorporate_subject = yes
			can_have_incorporated_vassals = yes
		}
		FROM = {
			is_subject_of = ROOT
			is_subject_of_type = vassal
		}
	}
	is_allowed = {
		is_at_war = no
		NOT = {
			incorporated_vassal = 5
		}
		NOT = {
			truce_with = FROM
		}
		FROM = {
			has_opinion = {
				who = ROOT
				value = 190
			}
		}
		FROM = {
			NOT = {
				liberty_desire = 25
			}
		}
		FROM = {
			years_in_vassalage_under = {
				who = ROOT
				years = 10
			}
		}
	}
	on_accept = {
		create_subject = {
			subject_type = incorporated_vassal
			subject = FROM
			keep_liberty_desire = yes
		}
		FROM = {
			capital_scope = {
				add_unrest = 2
			}
			add_liberty_desire = 10
			add_stability = -1
		}
	}
	on_decline = {
	}
	ai_will_do = {
		always = no
	}
}

consolidate_vassal = {
	category = influence
	require_acceptance = no
	is_visible = {
		can_consolidate_subject = yes
		FROM = {
			is_subject_of = ROOT
			is_subject_of_type = incorporated_vassal
		}
	}
	is_allowed = {
		is_at_war = no
		NOT = {
			consolidated_vassal = 5
		}
		NOT = {
			truce_with = FROM
		}
		FROM = {
			has_opinion = {
				who = ROOT
				value = 190
			}
		}
		FROM = {
			NOT = {
				liberty_desire = 25
			}
		}
	}
	on_accept = {
		create_subject = {
			subject_type = consolidated_vassal
			subject = FROM
			keep_liberty_desire = yes
		}
		FROM = {
			capital_scope = {
				add_unrest = 2
			}
			add_liberty_desire = 10
			add_stability = -1
		}
		FROM = {
			every_owned_province = {
				limit = {
					is_core = FROM
				}
				add_core = ROOT
			}
		}
	}
	on_decline = {
	}
	ai_will_do = {
		always = no
	}
}
