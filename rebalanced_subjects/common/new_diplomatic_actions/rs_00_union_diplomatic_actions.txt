convert_to_incorporated_personal_union = {
	category = influence
	require_acceptance = no
	is_visible = {
		adm_tech = 12		#Early Modern Administration
		NOT = {
			tag = HAB
		}
		OR = {
			FROM = {
				is_subject_of = ROOT
				is_subject_of_type = personal_union
			}
			FROM = {
				is_subject_of = ROOT
				is_subject_of_type = sovereign_personal_union
			}
		}
	}
	is_allowed = {
		dip_power = 50
		is_at_war = no
		has_regency = no
		has_heir = yes
		NOT = {
			truce_with = FROM
		}
		FROM = {
			is_at_war = no
		}
		FROM = {
			NOT = {
				truce_with = ROOT
			}
		}
		FROM = {
			years_in_union_under = {
				who = ROOT
				years = 15
			}
		}
		FROM = {
			has_opinion = {
				who = ROOT
				value = 190
			}
		}
		FROM = {
			NOT = {
				liberty_desire = 50
			}
		}
	}
	on_accept = {
		add_dip_power = -50
		create_subject = {
			subject_type = incorporated_personal_union
			subject = FROM
			keep_liberty_desire = yes
		}
		if = {
			limit = {
				FROM = {
					is_subject_of = ROOT
					is_subject_of_type = sovereign_personal_union
				}
			}
			FROM = {
				add_liberty_desire = 50
				add_opinion = {
					who = THIS
					modifier = sovereignty_broken
				}
			}
		}
	}
	on_decline = {
	}
	ai_will_do = {
		always = no
	}
}
