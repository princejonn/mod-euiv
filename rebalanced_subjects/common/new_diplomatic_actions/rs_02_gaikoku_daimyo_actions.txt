form_gaikoku_daimyo = {
	category = influence
	require_acceptance = no
	is_visible = {
		can_be_overlord = gaikoku_daimyo
		can_be_gaikoku_subject = yes
		FROM = {
			is_subject_of = ROOT
			is_any_vassal_type = yes
		}
	}
	is_allowed = {
		government_rank = 3
		is_at_war = no
		FROM = {
			has_opinion = {
				who = ROOT
				value = 100
			}
		}
		FROM = {
			NOT = {
				liberty_desire = 50
			}
		}
	}
	on_accept = {
		create_subject = {
			subject_type = gaikoku_daimyo
			subject = FROM
			keep_liberty_desire = yes
		}
		hidden_effect = {
			FROM = {
				if = {
					limit = {
						NOT = {
							government = monarchy
						}
					}
					change_government = monarchy
					add_government_reform = feudalism_reform
				}
				capital_scope = {
					add_unrest = 2
				}
				add_liberty_desire = 5
				add_stability = -1
			}
		}
	}
	on_decline = {
	}
	ai_will_do = {
		always = no
	}
}
