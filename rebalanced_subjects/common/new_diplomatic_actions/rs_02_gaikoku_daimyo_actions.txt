form_gaikoku_daimyo = {
	category = influence
	require_acceptance = no
	is_visible = {
		ai = no
		can_be_overlord = gaikoku_daimyo
		can_be_gaikoku_daimyo_subject = yes
		FROM = {
			is_subject_of = ROOT
			is_any_vassal_type = yes
		}
	}
	is_allowed = {
		government_rank = 3
		is_at_war = no
		FROM = {
			has_opinion = {
				who = ROOT
				value = 100
			}
		}
		FROM = {
			NOT = {
				liberty_desire = 50
			}
		}
	}
	on_accept = {
		create_subject = {
			subject_type = gaikoku_daimyo
			subject = FROM
			keep_liberty_desire = yes
		}
		create_gaikoku_daimyo = yes
		FROM = {
			capital_scope = {
				add_unrest = 2
			}
			add_liberty_desire = 10
			add_stability = -1
		}
	}
	on_decline = {
	}
	ai_will_do = {
		always = no
	}
}
