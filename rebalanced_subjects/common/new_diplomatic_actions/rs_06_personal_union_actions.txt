incorporate_personal_union = {
	category = influence
	require_acceptance = no
	is_visible = {
		ai = no
		OR = {
			can_incorporate_subject = yes
			has_country_flag = hab_integrated_vassals_flag
		}
		FROM = {
			is_subject_of = ROOT
			is_subject_of_type = personal_union
		}
	}
	is_allowed = {
		is_at_war = no
		NOT = {
			incorporated_personal_union = 5
		}
		NOT = {
			truce_with = FROM
		}
		FROM = {
			has_opinion = {
				who = ROOT
				value = 190
			}
		}
		FROM = {
			NOT = {
				liberty_desire = 25
			}
		}
		FROM = {
			years_in_union_under = {
				who = ROOT
				years = 50
			}
		}
	}
	on_accept = {
		create_subject = {
			subject_type = incorporated_personal_union
			subject = FROM
			keep_liberty_desire = yes
		}
		FROM = {
			capital_scope = {
				add_unrest = 2
			}
			add_liberty_desire = 10
			add_stability = -1
		}
	}
	on_decline = {
	}
	ai_will_do = {
		always = no
	}
}

consolidate_personal_union = {
	category = influence
	require_acceptance = no
	is_visible = {
		ai = no
		can_consolidate_subject = yes
		FROM = {
			is_subject_of = ROOT
			OR = {
				is_subject_of_type = incorporated_personal_union
				is_subject_of_type = integrated_personal_union
			}
		}
	}
	is_allowed = {
		is_at_war = no
		NOT = {
			consolidated_personal_union = 5
		}
		NOT = {
			truce_with = FROM
		}
		FROM = {
			has_opinion = {
				who = ROOT
				value = 190
			}
		}
		FROM = {
			NOT = {
				liberty_desire = 25
			}
		}
	}
	on_accept = {
		create_subject = {
			subject_type = consolidated_personal_union
			subject = FROM
			keep_liberty_desire = yes
		}
		FROM = {
			capital_scope = {
				add_unrest = 2
			}
			add_liberty_desire = 10
			add_stability = -1
		}
		FROM = {
			every_owned_province = {
				limit = {
					is_core = FROM
				}
				add_core = ROOT
			}
		}
	}
	on_decline = {
	}
	ai_will_do = {
		always = no
	}
}
