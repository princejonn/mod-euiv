form_principality = {
	category = influence
	require_acceptance = no
	is_visible = {
		government = monarchy
		FROM = {
			is_subject_of = ROOT
			OR = {
				# vassal
				is_subject_of_type = vassal
				is_subject_of_type = incorporated_vassal
				is_subject_of_type = consolidated_vassal
				# union
				is_subject_of_type = personal_union
				is_subject_of_type = incorporated_personal_union
				is_subject_of_type = consolidated_personal_union
				is_subject_of_type = integrated_personal_union
				is_subject_of_type = theodoro_personal_union
			}
		}
	}
	is_allowed = {
		government_rank = 3
		is_at_war = no
		full_idea_group = influence_ideas
		NOT = {
			truce_with = FROM
		}
		FROM = {
			has_opinion = {
				who = ROOT
				value = 190
			}
		}
		FROM = {
			NOT = {
				liberty_desire = 10
			}
		}
		OR = {
			FROM = {
				years_in_vassalage_under = {
					who = ROOT
					years = 10
				}
			}
			FROM = {
				years_in_union_under = {
					who = ROOT
					years = 25
				}
			}
		}
	}
	on_accept = {
		create_subject = {
			subject_type = principality
			subject = FROM
			keep_liberty_desire = yes
		}
		FROM = {
			if = {
				limit = {
					NOT = {
						government = monarchy
					}
				}
				change_government = monarchy
				add_government_reform = feudalism_reform
			}
			set_country_flag = has_overriden_color_flag
			change_country_color = {
				country = ROOT
			}
			define_ruler = {
				name = "Alexander"
				change_adm = 5
				change_dip = 5
				change_mil = 5
				culture = ROOT
				dynasty = ROOT
				religion = ROOT
				male = yes
				min_age = 30
				max_age = 50
			}
			define_heir = {
				max_random_adm = 6
				max_random_dip = 6
				max_random_mil = 6
				culture = ROOT
				dynasty = ROOT
				religion = ROOT
				male = yes
				age = 0
			}
			capital_scope = {
				add_unrest = 2
			}
			add_liberty_desire = 10
			add_stability = -1
		}
	}
	on_decline = {
	}
	ai_will_do = {
		always = no
	}
}
