designate_bishopric = {
	category = influence
	require_acceptance = no
	is_visible = {
		ai = no
		can_be_overlord = bishopric
		can_be_bishopric_subject = yes
		FROM = {
			is_subject_of = ROOT
			is_any_vassal_type = yes
		}
	}
	is_allowed = {
		is_at_war = no
		NOT = {
			bishopric = 3
		}
		FROM = {
			NOT = {
				num_of_cities = 4
			}
		}
	}
	on_accept = {
		create_subject = {
			subject_type = bishopric
			subject = FROM
			keep_liberty_desire = yes
		}
		hidden_effect = {
			FROM = {
				change_government = theocracy
				set_government_rank = 1
				if = {
					limit = {
						ROOT = {
							religion = orthodox
						}
					}
					add_government_reform = orthodox_patriarchate_reform
				}
				else = {
					add_government_reform = leading_clergy_reform
				}
				change_primary_culture = ROOT
				change_religion = ROOT
				random_owned_province = {
					limit = {
						NOT = {
							has_building = cathedral
						}
					}
					add_building = cathedral
				}
			}
		}
	}
	on_decline = {
	}
	ai_will_do = {
		always = no
	}
}

revoke_bishopric = {
	category = influence
	require_acceptance = no
	is_visible = {
		ai = no
		FROM = {
			is_subject_of = ROOT
			is_subject_of_type = bishopric
		}
	}
	is_allowed = {
		is_at_war = no
		FROM = {
			NOT = {
				liberty_desire = 50
			}
		}
	}
	on_accept = {
		create_subject = {
			subject_type = vassal
			subject = FROM
			keep_liberty_desire = yes
		}
		FROM = {
			add_liberty_desire = 25
			add_opinion = {
				who = ROOT
				modifier = broke_bishopric
			}
		}
	}
	on_decline = {
	}
	ai_will_do = {
		always = no
	}
}
