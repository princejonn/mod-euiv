form_personal_union_from_client_state = {
	category = influence
	require_acceptance = no
	is_visible = {
		FROM = {
			is_subject_of = ROOT
		}
		OR = {
			FROM = {
				is_subject_of_type = client_state
			}
			FROM = {
				is_subject_of_type = incorporated_client_state
			}
		}
	}
	is_allowed = {
		is_at_war = no
		has_regency = no
		marriage_with = FROM
		NOT = {
			truce_with = FROM
		}
		FROM = {
			is_at_war = no
		}
		FROM = {
			NOT = {
				truce_with = ROOT
			}
		}
		FROM = {
			marriage_with = ROOT
		}
		FROM = {
			years_in_vassalage_under = {
				who = ROOT
				years = 25
			}
		}
		FROM = {
			has_opinion = {
				who = ROOT
				value = 190
			}
		}
		FROM = {
			NOT = {
				liberty_desire = 25
			}
		}
		OR = {
			NOT = {
				has_country_flag = changed_subject_type_cooldown_flag
			}
			had_country_flag = {
				flag = changed_subject_type_cooldown_flag
				days = 3650
			}
		}
	}
	on_accept = {
		if = {
			limit = {
				FROM = {
					is_subject_of_type_with_overlord = {
						who = ROOT
						type = client_state
					}
				}
			}
			create_subject = {
				subject_type = personal_union
				subject = FROM
				keep_liberty_desire = yes
			}
		}
		else_if = {
			limit = {
				FROM = {
					is_subject_of_type_with_overlord = {
						who = ROOT
						type = incorporated_client_state
					}
				}
			}
			create_subject = {
				subject_type = incorporated_personal_union
				subject = FROM
				keep_liberty_desire = yes
			}
		}
		FROM = {
			capital_scope = {
				add_unrest = 2
			}
			add_liberty_desire = 10
			add_stability = -1
		}
		set_country_flag = changed_subject_type_cooldown_flag
	}
	on_decline = {
	}
	ai_will_do = {
		always = no
	}
}

		}

		always = no
