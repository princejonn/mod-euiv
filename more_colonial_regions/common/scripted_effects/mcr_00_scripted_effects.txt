cede_non_colonial_provinces_to_overlord = {
	every_country = {
		limit = {
			is_colonial_country = yes
		}
		save_event_target_as = province_owner
		overlord = {
			save_event_target_as = province_owner_overlord
		}
		every_owned_province = {
			limit = {
				is_city = yes
				NOT = {
					colonial_region = capital
				}
			}
			remove_claim = event_target:province_owner
			remove_core = event_target:province_owner
			add_core = event_target:province_owner_overlord
			cede_province = event_target:province_owner_overlord
		}
	}
}

cede_small_colonial_nations_to_overlord = {
	if = {
		limit = {
			has_global_flag = enabled_colonial_merging_global_flag
		}
		every_country = {
			limit = {
				ai = yes
				is_subject = yes
				is_country_with_colonial_subjects = yes
			}
			overlord = {
				save_event_target_as = province_owner_overlord
			}
			every_owned_province = {
				limit = {
					is_city = yes
					is_in_colonial_region = yes
				}
				remove_claim = event_target:province_owner
				remove_core = event_target:province_owner
				add_core = event_target:province_owner_overlord
				cede_province = event_target:province_owner_overlord
			}
			every_subject_country = {
				limit = {
					is_small_colonial_country = yes
				}
				save_event_target_as = province_owner
				every_owned_province = {
					limit = {
						is_city = yes
					}
					remove_claim = event_target:province_owner
					remove_core = event_target:province_owner
					add_core = event_target:province_owner_overlord
					cede_province = event_target:province_owner_overlord
				}
			}
		}
	}
}
