assassinate_ruler = {
	category = covert
	require_acceptance = no
	is_visible = {
		ai = no
		FROM = {
			ai = yes
		}
	}
	is_allowed = {
		full_idea_group = spy_ideas
		has_spy_network_in = {
			who = FROM
			value = 90
		}
		treasury = 500
	}
	on_accept = {
		add_spy_network_in = {
			who = FROM
			value = -90
		}
		add_treasury = -500
		random_list = {
			90 = {
				FROM = {
					kill_ruler = yes
					add_stability = -1
				}
				random = {
					chance = 10
					FROM = {
						add_opinion = {
							who = ROOT
							modifier = ruler_assassination
						}
						set_ai_attitude = {
							attitude = attitude_hostile
							who = ROOT
							locked = no
						}
						add_casus_belli = {
							target = ROOT
							type = cb_humiliate
							months = 240
						}
					}
				}
			}
			10 = {
				FROM = {
					add_opinion = {
						who = ROOT
						modifier = ruler_assassination_attempt
					}
				}
			}
		}
	}
	on_decline = {
	}
	ai_will_do = {
		always = no
	}
}

forge_claim_on_throne = {
	category = covert
	require_acceptance = no
	is_visible = {
		government = monarchy
		FROM = {
			government = monarchy
		}
	}
	is_allowed = {
		full_idea_group = spy_ideas
		has_spy_network_in = {
			who = FROM
			value = 85
		}
		treasury = 200
		FROM = {
			dynasty = ROOT
		}
		custom_trigger_tooltip = {
			tooltip = forge_claim_marriage_tooltip
			OR = {
				marriage_with = FROM
				FROM = {
					marriage_with = ROOT
				}
			}
		}
		custom_trigger_tooltip = {
			tooltip = forge_claim_heir_tooltip
			OR = {
				has_heir = no
				NOT = {
					heir_claim = 50
				}
			}
		}
		custom_trigger_tooltip = {
			tooltip = forge_claim_prestige_tooltip
			AND = {
				prestige = FROM
				NOT = {
					FROM = {
						prestige = ROOT
					}
				}
			}
		}
	}
	on_accept = {
		add_treasury = -200
		add_spy_network_in = {
			who = FROM
			value = -85
		}
		add_casus_belli = {
			type = cb_restore_personal_union
			target = FROM
			months = 240
		}
	}
	on_decline = {
	}
	ai_will_do = {
		always = no
	}
}
