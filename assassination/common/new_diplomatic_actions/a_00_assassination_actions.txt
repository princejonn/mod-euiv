assassinate_ruler = {
	category = covert
	require_acceptance = no
	is_visible = {
		ai = no
		government = monarchy
		FROM = {
			government = monarchy
			ai = yes
		}
		full_idea_group = spy_ideas
	}
	is_allowed = {
		employed_advisor = {
			type = spymaster
		}
		has_spy_network_in = {
			who = FROM
			value = 80
		}
		dip_power = 80
		treasury = 250
	}
	on_accept = {
		add_spy_network_in = {
			who = FROM
			value = -80
		}
		add_dip_power = -80
		add_treasury = -250
		FROM = {
			kill_ruler = yes
			add_stability = -1
		}
	}
	on_decline = {
	}
	ai_will_do = {
		always = no
	}
}

forge_claim_on_throne = {
	category = dynasty
	require_acceptance = no
	is_visible = {
		ai = no
		government = monarchy
		FROM = {
			government = monarchy
		}
		full_idea_group = spy_ideas
	}
	is_allowed = {
		employed_advisor = {
			type = spymaster
		}
		has_spy_network_in = {
			who = FROM
			value = 80
		}
		dip_power = 80
		treasury = 250
		FROM = {
			dynasty = ROOT
		}
		custom_trigger_tooltip = {
			tooltip = forge_claim_marriage_tooltip
			OR = {
				marriage_with = FROM
				FROM = {
					marriage_with = ROOT
				}
			}
		}
		heir_claim = 80
		custom_trigger_tooltip = {
			tooltip = forge_claim_heir_tooltip
			FROM = {
				OR = {
					has_heir = no
					AND = {
						has_heir = yes
						NOT = {
							heir_claim = 100
						}
					}
				}
			}
		}
		prestige = 80
		custom_trigger_tooltip = {
			tooltip = forge_claim_prestige_tooltip
			AND = {
				prestige = FROM
				NOT = {
					FROM = {
						prestige = ROOT
					}
				}
			}
		}
	}
	on_accept = {
		add_dip_power = -80
		add_treasury = -250
		add_prestige = -25
		add_spy_network_in = {
			who = FROM
			value = -80
		}
		add_casus_belli = {
			type = cb_restore_personal_union
			target = FROM
			months = 300
		}
	}
	on_decline = {
	}
	ai_will_do = {
		always = no
	}
}
